实例

package main

import "fmt"

func main() {
   var hello = "Hello"
   var world string
   world = "World"
   var i, j = 2, 3
   fmt.Println(hello+" "+world+"!", i, j)
   var length = len(hello)
   fmt.Println("hello len:", length)
}
  • 没有分号
  • 类型在变量后面，可自动推导
  • 左花括号换行会引起编译错误
